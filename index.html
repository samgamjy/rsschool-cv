function indexEqualsValue(a) {
  var start = 0, end = a.length;
  var neg = -1;
  while(start < end){
    var i = Math.floor((start + end) / 2);
    if (a[i] > i){
      end = i;
    } else if(a[i] < i) {
      start = i+1;
    } else {
      neg = i;
      end = i;
    }
  }
  return neg;
}
